140  /// <summary>
141  /// Performs cleanup on the dictionaries in either MakeRoom or Maintenance mode. Returns true if the goal was achieved, false if the cleanup was canceled because another cleanup was executing conurrently.
142  /// </summary>
143  /// <param name="mode"></param>
144  private bool Cleanup(CleanupMode mode) {
145  if (mode == CleanupMode.MakeRoom && byteCeiling < 1) return true; //We don't perform minimal cleanups unless a ceiling is specified.
146 
147  //We have to track removed items so we can fire events later.
148  List<KeyValuePair<T, int>> removed = CounterRemoved != null ? new List<KeyValuePair<T, int>>() : null;
149 
150  //We have to weak lock method-level, because otherwise a background thread could be cleaning when GetOrAdd is called, and we could have a deadlock
151  //With syncLock locked in GetOrAdd, waiting on cleanupLock, and Cleanup locked on cleanupLock, waiting on GetOrAdd.
152  //If we didn't have any method-level lock, we'd waste resources with simultaneous cleanup runs.
153  //sortLock is redundant with cleanupLock, but remains in case I decide to pullt the method level lock
154  if (!Monitor.TryEnter(cleanupLock)) return false; //Failed to lock, another thread is cleaning.
155  try {
156  //In high precision, we lock the entire long-running process
157  if (precision == EventCountingStrategy.ThreadingPrecision.Accurate) Monitor.Enter(syncLock);
158  try {
159  EventCounter[] counters;
160  //In fast mode, only lock for the copy and delete. We can sort outside after taking a snapshot.
161  //We wont remove newly added ones, but thats ok.
162  lock (syncLock) {
163  if (mode == CleanupMode.MakeRoom && bytesUsed < byteCeiling) return true; //Nothing to do, there is stil room
164  counters = new EventCounter[countersToKeys.Count];
165  countersToKeys.Keys.CopyTo(counters, 0); //Clone
166  }
167  //Lock for sorting so we have synchronized access to a.sortValue
168  lock (sortLock) {
169  //Pause values
170  for (int i = 0; i < counters.Length; i++) {
171  counters[i].sortValue = counters[i].GetValue();
172  }
173  //Sort lowest counters to the top using quicksort
174  Array.Sort<EventCounter>(counters, delegate(EventCounter a, EventCounter b) {
175  return a.sortValue - b.sortValue;
176  });
177  }
178  //Go back into lock
179  lock (syncLock) {
180  long removedBytes = 0;
181  long goal = byteCeiling / 10; //10% is our goal if we are removing minimally.
182  //Remove items
183  for (int i = 0; i < counters.Length; i++) {
184  EventCounter c = counters[i];
185  if (mode == CleanupMode.MakeRoom && removedBytes >= goal) return true; //Done, we hit our goal!
186  if (mode == CleanupMode.Maintenance && c.sortValue > 0) return true; //Done, We hit the end of the zeros
187  if (mode == CleanupMode.Maintenance && c.GetValue() > 0) continue; //Skip counters that incremeted while we were working.
188  //Look up key
189  T key;
190  countersToKeys.TryGetValue(c, out key);
191  if (key.Equals(default(T))) continue; //Skip counters that have already been removed
192  //Remove counter
193  countersToKeys.Remove(c);
194  keysToCounters.Remove(key);
195  //Increment our local byte removal counter
196  removedBytes += c.CustomSize + itemSize;
197  //Decrement global counter
198  bytesUsed = bytesUsed - c.CustomSize - itemSize;
199  //store removed keys
200  if (removed != null) removed.Add(new KeyValuePair<T, int>(key, c.GetValue()));
201  }
202  }
203  } finally {
204  if (precision == EventCountingStrategy.ThreadingPrecision.Accurate) Monitor.Exit(syncLock);
205  }
206  } finally {
207  Monitor.Exit(cleanupLock);
208 
209  //Fire CounterRemoved event (may still be in syncLock)!
210  if (removed != null && this.CounterRemoved != null) {
211  foreach (KeyValuePair<T, int> p in removed) {
212  CounterRemoved(this, p.Key, p.Value);
213  }
214  }
215  }

218  return true;
