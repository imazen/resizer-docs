75  [SuppressMessage("Microsoft.Performance", "CA1822:MarkMembersAsStatic", Justification = "Intended to be member method.")]
76  public string Execute(GhostscriptSettings settings)
77  {
78  if(settings == null)
79  {
80  throw new ArgumentNullException("settings");
81  }
82 
83  // Flatten settings into Ghostscript arguments
84  string[] arguments = GetArguments(settings);
85 
86  // Use callback handlers to capture stdout and stderr.
87  // NOTE: Delegates do not need to be pinned.
88  StringBuilder outputBuilder = new StringBuilder();
89  StringBuilder errorBuilder = new StringBuilder();
90  GhostscriptMessageEventHandler outputHandler = (i, s, l) => HandleOutputMessage(outputBuilder, s, l);
91  GhostscriptMessageEventHandler errorHandler = (i, s, l) => HandleOutputMessage(errorBuilder, s, l);
92  GhostscriptMessageEventHandler inputHandler = (i, s, l) => l;
93 
94  // NOTE: Ghostscript supports only one instance per process
95  int result;
96  lock(_syncObject)
97  {
98  // Create a new instance of Ghostscript. This instance is passed to most other gsapi functions.
99  // The caller_handle will be provided to callback functions.
100  IntPtr instance;
101  GhostscriptNativeMethods.NewInstance(out instance, IntPtr.Zero);
102 
103  // Set the callback functions for stdio.
104  GhostscriptNativeMethods.SetMessageHandlers(instance, inputHandler, outputHandler, errorHandler);
105 
106  // Initialise the interpreter.
107  // This calls gs_main_init_with_args() in imainarg.c. See below for return codes.
108  // The arguments are the same as the "C" main function: argv[0] is ignored and the user supplied arguments are
109  // argv[1] to argv[argc-1].
110  result = GhostscriptNativeMethods.InitializeWithArguments(instance, arguments.Length, arguments);
111 
112  // Exit the interpreter.
113  // This must be called on shutdown if gsapi_init_with_args() has been called, and just before gsapi_delete_instance().
114  GhostscriptNativeMethods.Exit(instance);
115 
116  // Destroy an instance of Ghostscript.
117  // Before you call this, Ghostscript must have finished.
118  // If Ghostscript has been initialised, you must call gsapi_exit before gsapi_delete_instance.
119  GhostscriptNativeMethods.DeleteInstance(instance);
120  }
121 
122  // Check for errors. Zero and e_Quit(-101) are not errors.
123  string output = outputBuilder.ToString();
124  if(result != 0 && result != -101)
125  {
126  // Use error as message if output is empty
127  string error = errorBuilder.ToString();
128  if(string.IsNullOrEmpty(output))
129  {
130  output = error;
131  }
132 
133  GhostscriptException exception = new GhostscriptException(output);
134  exception.Data["args"] = arguments;
135  exception.Data["stderr"] = error;
136  throw exception;
137  }
138 
139  // Return the output message
140  return output;
141  }

143  private static string[] GetArguments(GhostscriptSettings settings)
144  {
145  // Flatten Ghostscript engine settings into string array
146  // - first argument ignore (argv[0])
147  // - input path as last argument
148  string[] arguments;
149  List<string> argumentItems = new List<string> { string.Empty };
150  foreach(string key in settings.AllKeys)
151  {
152  string[] values = settings.GetValues(key);
153  if(values == null)
154  {
155  // No value
156  argumentItems.Add(key);
157  }
158  else
159  {
160  IEnumerable<string> items = values.Select(value => string.Concat(key, value));
161  argumentItems.AddRange(items);
162  }
163  }
164  arguments = argumentItems.ToArray();
165  return arguments;
166  }

168  [SuppressMessage("Microsoft.Security", "CA2122:DoNotIndirectlyExposeMethodsWithLinkDemands", Justification = "Security not exposed.")]
169  private static Version GetNativeVersion()
170  {
171  // Get the Ghostscript version.
172  // GPL Ghostscript
173  // Copyright Some Company
174  // 815
175  // 20040922
176  // Extracted as a structure but we're only interested in the 3rd field (815).
177  GhostscriptNativeMethods.GhostscriptVersion ghostscriptVersion = new GhostscriptNativeMethods.GhostscriptVersion();
178  try
179  {
180  GhostscriptNativeMethods.GetRevision(ref ghostscriptVersion, Marshal.SizeOf(ghostscriptVersion));
181  }
182  catch(DllNotFoundException)
183  {
184  return null;
185  }
186  catch(MissingMethodException)
187  {
188  return null;
189  }
190 
191  // Convert version from integer such as '914' to Version object 9.14
192  string versionText = ((double)ghostscriptVersion.Revision / 100).ToString(CultureInfo.InvariantCulture);
193  return new Version(versionText);
194  }

196  public int HandleOutputMessage(StringBuilder builder, IntPtr source, int length)
197  {
198  string text;
199  if(length > 0)
200  {
201  // Marshal unmanaged output to local buffer
202  byte[] buffer = new byte[length];
203  Marshal.Copy(source, buffer, 0, buffer.Length);
204 
205  // Determine encoding style
206  System.Text.Encoding encoding = System.Text.Encoding.ASCII;
207  if(length > 2)
208  {
209  // Try to detect if UNICODE by looking for byte order mark and assigning appropriate encoder
210  if(buffer[0] == 0xFE && buffer[1] == 0xFF)
211  {
212  encoding = System.Text.Encoding.BigEndianUnicode;
213  }
214  else if(buffer[0] == 0xFF && buffer[1] == 0xFE)
215  {
216  encoding = System.Text.Encoding.Unicode;
217  }
218  }
219 
220  // Perform decode
221  text = encoding.GetString(buffer, 0, buffer.Length);
222  builder.Append(text);
223  }
224  return length;
225  }
