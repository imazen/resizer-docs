00199         /// <summary>
00200         /// Utility routine for transforming path names from filesystem format (on Windows that means backslashes) to
00201         /// a format suitable for use within zipfiles. This means trimming the volume letter and colon (if any) And
00202         /// swapping backslashes for forward slashes.

00206         public static string NormalizePathForUseInZipFile(string pathName)
00207         {
00208             // boundary case
00209             if (String.IsNullOrEmpty(pathName)) return pathName;
00210 
00211             // trim volume if necessary
00212             if ((pathName.Length >= 2) && ((pathName[1] == ':') && (pathName[2] == '\\')))
00213                 pathName = pathName.Substring(3);
00214 
00215             // swap slashes
00216             pathName = pathName.Replace('\\', '/');
00217 
00218             // trim all leading slashes
00219             while (pathName.StartsWith("/")) pathName = pathName.Substring(1);
00220 
00221             return SimplifyFwdSlashPath(pathName);
