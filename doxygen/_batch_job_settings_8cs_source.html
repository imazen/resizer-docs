163  /// <summary>
164  /// Loops through all files, assigning targetFilenames if they are null,
165  /// and eliminating duplicate names by adding _1, _2, etc.
166  /// Also normalizes filenames for use in zip folder.
167  /// Used internally. May be used externally also if calling code wishes to to know the final file names.
168  /// </summary>
169  public void FixDuplicateFilenames(string prefix = "_")
170  {
171  HashSet<string> names = new HashSet<string>(StringComparer.OrdinalIgnoreCase);
172  foreach (BatchResizeItem i in files)
173  {
174  //Assign default if null
175  if (i.TargetFilename == null) i.TargetFilename = System.IO.Path.GetFileNameWithoutExtension(i.PhysicalPath);
176 
177  //See if it is a duplicate
178  if (names.Contains(i.TargetFilename))
179  {
180  //Generate a unique filename by appending a number.
181  int index = 1;
182  string next = i.TargetFilename + prefix + index;
183  while (names.Contains(next))
184  {
185  index++;
186  next = i.TargetFilename + prefix + index;
187  }
188  i.TargetFilename = next;
189  }
190 
191  //Normalize for zip usage
192  i.TargetFilename = NormalizePathForUseInZipFile(i.TargetFilename);
193 
194  //Add to names set, so we can verify future filename uniqueness.
195  names.Add(i.TargetFilename);
196 
197  }
198  }
199  /// <summary>
200  /// Utility routine for transforming path names from filesystem format (on Windows that means backslashes) to
201  /// a format suitable for use within zipfiles. This means trimming the volume letter and colon (if any) And
202  /// swapping backslashes for forward slashes.
203  /// </summary>
204  /// <param name="pathName">source path.</param>
205  /// <returns>transformed path</returns>
206  public static string NormalizePathForUseInZipFile(string pathName)
207  {
208  // boundary case
209  if (String.IsNullOrEmpty(pathName)) return pathName;
210 
211  // trim volume if necessary
212  if ((pathName.Length >= 2) && ((pathName[1] == ':') && (pathName[2] == '\\')))
213  pathName = pathName.Substring(3);
214 
215  // swap slashes
216  pathName = pathName.Replace('\\', '/');
217 
218  // trim all leading slashes
219  while (pathName.StartsWith("/")) pathName = pathName.Substring(1);
220 
221  return SimplifyFwdSlashPath(pathName);
222  }
223 
224  private static string SimplifyFwdSlashPath(string path)
225  {
226  if (path.StartsWith("./")) path = path.Substring(2);
227  path = path.Replace("/./", "/");
228  // Replace foo/anything/../bar with foo/bar
229  var re = new System.Text.RegularExpressions.Regex(@"^(.*/)?([^/\\.]+/\\.\\./)(.+)$");
230  path = re.Replace(path, "$1$3");
231  return path;
232  }
