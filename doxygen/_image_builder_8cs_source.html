00628         private ImageColorFormat GetColorFormat(Bitmap bitmap)
00629         {
00630             const int pixelFormatIndexed = 0x00010000;
00631             const int pixelFormat32bppCMYK = 0x200F;
00632             const int pixelFormat16bppGrayScale = (4 | (16 << 8));
00633 
00634             // Check image flags
00635             var flags = (ImageFlags)bitmap.Flags;
00636             if ((flags & ImageFlags.ColorSpaceCmyk) > 0 || (flags & (ImageFlags.ColorSpaceYcck)) > 0)
00637             {
00638                 return ImageColorFormat.Cmyk;
00639             }
00640             else if ((flags & ImageFlags.ColorSpaceGray) > 0)
00641             {
00642                 return ImageColorFormat.Grayscale;
00643             }
00644 
00645             // Check pixel format
00646             var pixelFormat = (int)bitmap.PixelFormat;
00647             if (pixelFormat == pixelFormat32bppCMYK)
00648             {
00649                 return ImageColorFormat.Cmyk;
00650             }
00651             else if ((pixelFormat & pixelFormatIndexed) != 0)
00652             {
00653                 return ImageColorFormat.Indexed;
00654             }
00655             else if (pixelFormat == pixelFormat16bppGrayScale)
00656             {
00657                 return ImageColorFormat.Grayscale;
00658             }
00659 
00660             // Default to RGB
00661             return ImageColorFormat.Rgb;
00662         }

00951         /// <param name="s"></param>
00952         protected override RequestedAction LayoutImage(ImageState s) {
00953             if (base.LayoutImage(s) == RequestedAction.Cancel) return RequestedAction.Cancel; //Call extensions
00954 
00955             if (s.copyRect.IsEmpty) {
00956                 //Use the crop size if present.
00957                 s.copyRect = new RectangleF(new PointF(0, 0), s.originalSize);
00958                 if (NameValueCollectionExtensions.GetList<double>(s.settings, "crop", 0, 4) != null) {
00959                     s.copyRect = PolygonMath.ToRectangle(s.settings.getCustomCropSourceRect(s.originalSize)); //Round the custom crop rectangle coordinates
00960                     if (s.copyRect.Size.IsEmpty) throw new Exception("You must specify a custom crop rectange if crop=custom");
00961                 }
00962             }
00963             //Save the manual crop size.
00964             SizeF manualCropSize = s.copySize;
00965             RectangleF manualCropRect = s.copyRect;
00966 
00967             FitMode fit = s.settings.Mode;
00968             //Determine fit mode to use if both vertical and horizontal limits are used.
00969             if (fit == FitMode.None){
00970                 if (s.settings.Width != -1 || s.settings.Height != -1){
00971 
00972                     if ("fill".Equals(s.settings["stretch"], StringComparison.OrdinalIgnoreCase)) fit = FitMode.Stretch;
00973                     else if ("auto".Equals(s.settings["crop"], StringComparison.OrdinalIgnoreCase)) fit = FitMode.Crop;
00974                     else if (!string.IsNullOrEmpty(s.settings["carve"]) 
00975                         && !"false".Equals(s.settings["carve"], StringComparison.OrdinalIgnoreCase)
00976                         && !"none".Equals(s.settings["carve"], StringComparison.OrdinalIgnoreCase)) fit = FitMode.Carve;
00977                     else fit = FitMode.Pad;
00978                 }else{
00979                     fit = FitMode.Max;
00980                 }
00981 
00982             }
00983             
00984 
00985 
00986 
00987             //Aspect ratio of the image
00988             double imageRatio = s.copySize.Width / s.copySize.Height;
00989 
00990             //Zoom factor
00991             double zoom = s.settings.Get<double>("zoom", 1);
00992 
00993             //The target size for the image 
00994             SizeF targetSize = new SizeF(-1, -1);
00995             //Target area for the image
00996             SizeF areaSize = new SizeF(-1, -1);
00997             //If any dimensions are specified, calculate. Otherwise, use original image dimensions
00998             if (s.settings.Width != -1 || s.settings.Height != -1 || s.settings.MaxHeight != -1 || s.settings.MaxWidth != -1) {
00999                 //A dimension was specified. 
01000                 //We first calculate the largest size the image can be under the width/height/maxwidth/maxheight restrictions.
01001                 //- pretending stretch=fill and scale=both
01002 
01003                 //Temp vars - results stored in targetSize and areaSize
01004                 double width = s.settings.Width;
01005                 double height = s.settings.Height;
01006                 double maxwidth = s.settings.MaxWidth;
01007                 double maxheight = s.settings.MaxHeight;
01008 
01009                 //Eliminate cases where both a value and a max value are specified: use the smaller value for the width/height 
01010                 if (maxwidth > 0 && width > 0) {   width = Math.Min(maxwidth, width);    maxwidth = -1;  }
01011                 if (maxheight > 0 && height > 0) { height = Math.Min(maxheight, height); maxheight = -1; }
01012 
01013                 //Handle cases of width/maxheight and height/maxwidth as in legacy versions. 
01014                 if (width != -1 && maxheight != -1) maxheight = Math.Min(maxheight, (width / imageRatio));
01015                 if (height != -1 && maxwidth != -1) maxwidth = Math.Min(maxwidth, (height * imageRatio));
01016 
01017 
01018                 //Move max values to width/height. FitMode should already reflect the mode we are using, and we've already resolved mixed modes above.
01019                 width = Math.Max(width, maxwidth);
01020                 height = Math.Max(height, maxheight);
01021

01032                 } else if (fit == FitMode.Pad) {

01038                     areaSize = targetSize = new SizeF((float)width, (float)height);

01041                     //Center the portion we are copying within the manualCropSize
01042                     s.copyRect = PolygonMath.ToRectangle(PolygonMath.AlignWith(new RectangleF(0, 0, sourceSize.Width, sourceSize.Height), s.copyRect, s.settings.Anchor));
01043 
01044                 } else { //Stretch and carve both act like stretching, so do that:
01045                     areaSize = targetSize = new SizeF((float)width, (float)height);
01046                 }
01047 
01048 
01049             }else{

01051                 areaSize = targetSize = manualCropSize; 
01052             }
01053 
01054             //Multiply both areaSize and targetSize by zoom. 
01055             areaSize.Width *= (float)zoom;
01056             areaSize.Height *= (float)zoom;

01058             targetSize.Height *= (float)zoom;

01060             //Todo: automatic crop is permitted to break the scaling rules. Fix!!
01061 
01062             //Now do upscale/downscale checks. If they take effect, set targetSize to imageSize
01063             if (s.settings.Scale == ScaleMode.DownscaleOnly) {
01064                 if (PolygonMath.FitsInside(manualCropSize, targetSize)) {
01065                     //The image is smaller or equal to its target polygon. Use original image coordinates instead.
01066                     areaSize = targetSize = manualCropSize;
01067                     s.copyRect = manualCropRect;
01068                 }
01069             } else if (s.settings.Scale == ScaleMode.UpscaleOnly) {
01070                 if (!PolygonMath.FitsInside(manualCropSize, targetSize)) {
01071                     //The image is larger than its target. Use original image coordintes instead
01072                     areaSize = targetSize = manualCropSize;
01073                     s.copyRect = manualCropRect;
01074                 }
01075             } else if (s.settings.Scale == ScaleMode.UpscaleCanvas) {
01076                 //Same as downscaleonly, except areaSize isn't changed.
01077                 if (PolygonMath.FitsInside(manualCropSize, targetSize)) {
01078                     //The image is smaller or equal to its target polygon. 
01079                     
01080                     //Use manual copy rect/size instead.
01081 
01082                     targetSize = manualCropSize;
01083                     s.copyRect = manualCropRect;
01084                 }
01085             }
01086 
01087 
01088             //May 12: require max dimension and round values to minimize rounding differences later.
01089             areaSize.Width = Math.Max(1, (float)Math.Round(areaSize.Width));
01090             areaSize.Height = Math.Max(1, (float)Math.Round(areaSize.Height));
01091             targetSize.Width = Math.Max(1, (float)Math.Round(targetSize.Width));
01092             targetSize.Height = Math.Max(1, (float)Math.Round(targetSize.Height));
01093             
01094 
01095             //Translate and scale all existing rings
01096             s.layout.Shift(s.copyRect, new RectangleF(new Point(0, 0), targetSize));
01097 
01098             s.layout.AddRing("image", PolygonMath.ToPoly(new RectangleF(new PointF(0, 0), targetSize)));
01099 
01100             s.layout.AddRing("imageArea",PolygonMath.ToPoly(new RectangleF(new PointF(0, 0), areaSize)));
01101 
01102             //Center imageArea around 'image'
01103             s.layout["imageArea"] = PolygonMath.AlignWith(s.layout["imageArea"], s.layout["image"], s.settings.Anchor);
01104 
01105             return RequestedAction.None;
