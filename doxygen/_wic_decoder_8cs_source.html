63  var streamWrapper = factory.CreateStream();
64  streamWrapper.InitializeFromMemory(data, (uint)lData);
65  var decoder = factory.CreateDecoderFromStream(streamWrapper, null,
66  WICDecodeOptions.WICDecodeMetadataCacheOnLoad);
67 
68  try {
69 
70  //Figure out which frame to work with
71  int frameIndex = 0;
72  if (!string.IsNullOrEmpty(settings["page"]) && !int.TryParse(settings["page"], NumberStyles.Number, NumberFormatInfo.InvariantInfo, out frameIndex))
73  if (!string.IsNullOrEmpty(settings["frame"]) && !int.TryParse(settings["frame"], NumberStyles.Number, NumberFormatInfo.InvariantInfo, out frameIndex))
74  frameIndex = 0;
75

78 
79  if (frameIndex > 0) {

83  frameIndex = Math.Max(0, frameIndex);
84 
85  if ("true".Equals(settings["usepreview"]))

87  else if ("true".Equals(settings["usethumb"]))
88  frame = decoder.GetThumbnail();
89  else

93  GC.KeepAlive(data);
