152  private byte[] Equalize(int[] histogram, long totalPixels) {
153 
154  //Low and high indexes to stretch
155  int low = 0; int high = 255;
156 
157  if (Algorithm == HistogramThresholdAlgorithm.Area)
158  AreaThreshold(histogram, (double)totalPixels, LowThreshold, HighThreshold, out low, out high);
159  else if (Algorithm == HistogramThresholdAlgorithm.GIMP)
160  GIMPThreshold(histogram, (double)totalPixels, LowThreshold, HighThreshold, out low, out high);
161  else
162  SimpleThreshold(histogram, (double)totalPixels, LowThreshold, HighThreshold, out low, out high);
163

165  double scale = 255.0 / (double)(high - low);
166 
167  //Create the new, scaled mapping
168  byte[] equalizedHistogram = new byte[256];
169  for (int i = 0; i < 256; i++) {

171  }
