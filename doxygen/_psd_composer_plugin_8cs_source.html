269  /// <summary>
270  /// Mixes the specified color (using the included alpha value as a weight) into the bitmap, leaving the bitmap's original alpha values untouched.
271  /// </summary>
272  /// <param name="b"></param>
273  /// <param name="c"></param>
274  private void ColorBitmap(Bitmap b, Color c)
275  {
276  double weightedR = c.R * c.A;
277  double weightedG = c.G * c.A;
278  double weightedB = c.B * c.A;
279  double originalWeight = 255 - c.A;
280 
281  int width = b.Width; int height = b.Height;
282  BitmapData bd = b.LockBits(new Rectangle(0,0,width,height), ImageLockMode.ReadWrite, b.PixelFormat);
283  unsafe
284  {
285  byte* pCurrRowPixel = (byte*)bd.Scan0.ToPointer();
286  for (int y = 0; y < height; y++)
287  {
288  int rowIndex = y * width;
289  PhotoshopFile.ImageDecoder.PixelData* pCurrPixel = (PhotoshopFile.ImageDecoder.PixelData*)pCurrRowPixel;
290  for (int x = 0; x < width; x++)
291  {
292  pCurrPixel->Red = (byte)(((double)pCurrPixel->Red * originalWeight + weightedR) / 255);
293  pCurrPixel->Green = (byte)(((double)pCurrPixel->Green * originalWeight + weightedG) / 255);
294  pCurrPixel->Blue = (byte)(((double)pCurrPixel->Blue * originalWeight + weightedB) / 255);
295  pCurrPixel += 1;
296  }
297  pCurrRowPixel += bd.Stride;
298  }
299  }
300 
301  b.UnlockBits(bd);
302  }

304  private ComposeLayerDelegate BuildModifyLayerCallback(NameValueCollection queryString)
305  {
306  PsdCommandSearcher searcher = new PsdCommandSearcher(new PsdCommandBuilder(queryString));
307 
308  return delegate(Graphics g, Bitmap b, object layer)
309  {
310  PhotoshopFile.Layer l = (PhotoshopFile.Layer)layer;
311  //See if this layer is supposed to be re-colored.
312  Nullable<Color> color = searcher.getColor(l.Name);
313 
314  if (b == null && l.Rect.X == 0 && l.Rect.Y == 0 && l.Rect.Width == 0 && l.Rect.Height == 0)
315  {
316  return; //This layer has no size, it is probably a layer group.
317  }
318 
319  //See if we need to re-draw this text layer
320  Nullable<bool> redraw = searcher.getRedraw(l.Name);
321  if (redraw != null && redraw == true)
322  {
323  //Verify it has text layer information:
324  bool hasText = false;
325  foreach (PhotoshopFile.Layer.AdjustmentLayerInfo lInfo in l.AdjustmentInfo)
326  if (lInfo.Key.Equals("TySh", StringComparison.Ordinal)){ hasText=true; break;}
327 
328 
329  if (hasText) {
330  //Re-draw the text directly, ignoring the bitmap
331  var tlr = new TextLayerRenderer(l);
332  tlr.IgnoreMissingFonts = !isStrictMode();
333  tlr.Render(g, color, searcher.getReplacementText(l.Name));
334  return;
335  }
336  }
337 
338 
339  if (b == null && !isStrictMode()) return; //Skip drawing layers that have no bitmap data.
340  if (b == null) throw new Exception("No bitmap data found for layer " + l.Name);
341  //Draw the existing bitmap
342  //Blend color into bitmap
343  if (color != null) ColorBitmap(b, color.Value);
344  //Draw image
345  g.DrawImage(b, l.Rect);
346 
347  };
348  }
