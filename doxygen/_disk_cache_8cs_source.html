255  if (!IsConfigurationValid()) return false;
256  lock (_startSync) {
257  if (_started) return true;
258  if (!IsConfigurationValid()) return false;
259 
260  //Init the writer.
261  writer = new WebConfigWriter(this,PhysicalCacheDir);
262  //Init the inner cache
263  cache = new CustomDiskCache(this, PhysicalCacheDir, Subfolders, HashModifiedDate,AsyncBufferSize);
264  //Init the cleanup strategy
265  if (AutoClean && cleanupStrategy == null) cleanupStrategy = new CleanupStrategy(); //Default settings if null
266  //Init the cleanup worker
267  if (AutoClean) cleaner = new CleanupManager(this, cache, cleanupStrategy);
268  //If we're running with subfolders, enqueue the cache root for cleanup (after the 5 minute delay)
269  //so we don't eternally 'skip' files in the root or in other unused subfolders (since only 'accessed' subfolders are ever cleaned ).
270  if (cleaner != null) cleaner.CleanAll();
271 
272  if (log != null) log.Info("DiskCache started successfully.");
273  //Started successfully
274  _started = true;
275  return true;
276  }

295  public void Process(HttpContext context, IResponseArgs e) {

297  CacheResult r = Process(e);
298  context.Items["FinalCachedFile"] = r.PhysicalPath;

304

311  };

316  public CacheResult Process(IResponseArgs e) {
317  //Query for the modified date of the source file. If the source file changes on us during the write,
318  //we (may) end up saving the newer version in the cache properly, but with an older modified date.
319  //This will not cause any actual problems from a behavioral standpoint - the next request for the
320  //image will cause the file to be overwritten and the modified date updated.
321  DateTime modDate = e.HasModifiedDate ? e.GetModifiedDateUTC() : DateTime.MinValue;
322 
323  //Verify web.config exists in the cache folder.
324  writer.CheckWebConfigEvery5();
325 
326  //Cache the data to disk and return a path.
327  CacheResult r = cache.GetCachedFile(e.RequestKey, e.SuggestedExtension, e.ResizeImageToStream, modDate, CacheAccessTimeout,AsyncWrites);
328 
329  //Fail
330  if (r.Result == CacheQueryResult.Failed)
331  throw new ImageResizer.ImageProcessingException("Failed to acquire a lock on file \"" + r.PhysicalPath + "\" within " + CacheAccessTimeout + "ms. Caching failed.");
332 
333  if (r.Result == CacheQueryResult.Hit && cleaner != null)
334  cleaner.UsedFile(r.RelativePath, r.PhysicalPath);
335 
336  return r;
337  }
