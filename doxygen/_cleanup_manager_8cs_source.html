28  this.AddedFile(r.RelativePath); //It was either updated or added.

30  this.BeLazy();

33  cache.Index.FileDisappeared += delegate(string relativePath, string physicalPath) {
34  if (lp.Logger != null) lp.Logger.Warn("File disappeared from the cache unexpectedly - reindexing entire cache. File name: {0}", relativePath);
35  //Stop everything ASAP and start a brand new cleaning run.
36  queue.ReplaceWith(new CleanupWorkItem(CleanupWorkItem.Kind.CleanFolderRecursive, "", cache.PhysicalCachePath));
37  worker.MayHaveWork();
38  };

40  worker = new CleanupWorker(lp, cs,queue,cache);
41  }
