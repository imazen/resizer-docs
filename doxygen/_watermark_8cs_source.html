00066                 string name = c.Attrs["name"];

00072                         "XML: " + c.ToString(), IssueSeverity.ConfigurationError));
00073                         continue;
00074                     }
00075                 }
00076 
00077                 
00078 
00079                 if (c.Name.Equals("otherimages", StringComparison.OrdinalIgnoreCase)) otherImageDefaults = new ImageLayer(c.Attrs, this.c);
00080                 if (c.Name.Equals("image", StringComparison.OrdinalIgnoreCase)) dict.Add(name, new Layer[]{new ImageLayer(c.Attrs, this.c)});
00081                 if (c.Name.Equals("text", StringComparison.OrdinalIgnoreCase)) dict.Add(name, new Layer[] {new TextLayer(c.Attrs) });
00082                 if (c.Name.Equals("group", StringComparison.OrdinalIgnoreCase)) {
00083                     
00084                     List<Layer> layers = new List<Layer>();
00085                     if (c.Children != null) {
00086                         foreach (Node layer in c.Children) {
00087                             if (layer.Name.Equals("image", StringComparison.OrdinalIgnoreCase)) layers.Add(new ImageLayer(layer.Attrs, this.c));
00088                             if (layer.Name.Equals("text", StringComparison.OrdinalIgnoreCase)) layers.Add(new TextLayer(layer.Attrs));
00089                         }
00090                     }
00091                     dict.Add(name, layers);
00092                 }  
00093             }
00094             return dict;
