63  protected Dictionary<string, IEnumerable<Layer>> ParseWatermarks(Node n, ref ImageLayer otherImageDefaults) {
64  Dictionary<string, IEnumerable<Layer>> dict = new Dictionary<string, IEnumerable<Layer>>(StringComparer.OrdinalIgnoreCase);
65  if (n == null || n.Children == null) return dict;
66  foreach (Node c in n.Children) {
67  //Verify the name is specified and is unique.
68  string name = c.Attrs["name"];
69  if (c.Name.Equals("image", StringComparison.OrdinalIgnoreCase)
70  || c.Name.Equals("text", StringComparison.OrdinalIgnoreCase)
71  || c.Name.Equals("group", StringComparison.OrdinalIgnoreCase)) {
72  if (string.IsNullOrEmpty(name) || dict.ContainsKey(name)) {
73  this.c.configurationSectionIssues.AcceptIssue(new Issue("WatermarkPlugin", "The name attribute for each watermark or watermark group must be specified, and must be unique.",
74  "XML: " + c.ToString(), IssueSeverity.ConfigurationError));
75  continue;
76  }
77  }
78 
79 
80 
81  if (c.Name.Equals("otherimages", StringComparison.OrdinalIgnoreCase)) otherImageDefaults = new ImageLayer(c.Attrs, this.c);
82  if (c.Name.Equals("image", StringComparison.OrdinalIgnoreCase)) dict.Add(name, new Layer[]{new ImageLayer(c.Attrs, this.c)});
83  if (c.Name.Equals("text", StringComparison.OrdinalIgnoreCase)) dict.Add(name, new Layer[] {new TextLayer(c.Attrs) });
84  if (c.Name.Equals("group", StringComparison.OrdinalIgnoreCase)) {
85 
86  List<Layer> layers = new List<Layer>();
87  if (c.Children != null) {
88  foreach (Node layer in c.Children) {
89  if (layer.Name.Equals("image", StringComparison.OrdinalIgnoreCase)) layers.Add(new ImageLayer(layer.Attrs, this.c));
90  if (layer.Name.Equals("text", StringComparison.OrdinalIgnoreCase)) layers.Add(new TextLayer(layer.Attrs));
91  }
92  }
93  dict.Add(name, layers);
94  }
95  }
96  return dict;
97  }
