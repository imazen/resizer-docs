44  internal static void NotifyUse(Config config, Guid feature) {
45  //We only deal with ASP.NET requests. Outside of an HTTP request, no licensing is enforced
46  if (HttpContext.Current == null) return;
47 
48  //Map local requests if configured
49  string domain = HttpContext.Current.Request.Url.DnsSafeHost;
50  if (HttpContext.Current.Request.IsLocal) {
51  domain = config.get("licenses.local.use", null);
52  //We can't continue if we don't have a domain to work with
53  if (domain == null) return;
54  }
55  //Locate master license service
56  ILicenseService s = config.Plugins.GetOrInstall<ILicenseService>(CreateVerifier());
57 
58  //Verify the ILicenseService instance hasn't been replaced or tampered with since startup
59  lock(lockTracker){
60  long seed = feature.ToString().GetHashCode();
61  long auth = seed;
62  if (tracker == null) tracker = new Dictionary<Config, Dictionary<Guid, long>>();
63  Dictionary<Guid, long> ct;
64  if (!tracker.TryGetValue(config, out ct)) {
65  ct = new Dictionary<Guid, long>();
66  tracker.Add(config, ct);
67  } else if (!ct.TryGetValue(feature, out auth)) auth = seed;
68  long response = s.VerifyAuthenticity(feature);
69  if (response != auth) throw new ImageProcessingException("Licensing service responded incorrectly; tampering suspected.");
70  tracker[config][feature] = auth + 1;
71  }
72  //Inform the license service that the specified feature is in use for the given domain and configuration
73  s.NotifyUse(domain, feature);
74  }
