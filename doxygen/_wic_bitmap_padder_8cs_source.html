104  if (prc != null) {
105  left -= prc.X;
106  top -= prc.Y;
107  right -= (int)(fullWidth - prc.X - prc.Width);
108  bottom -= (int)(fullHeight - prc.Y - prc.Height);
109  }

111  //Now, adjust negative padding to crop from the source bitmap. We should end up with only positive padding.
112  if (left < 0) { crop.X += left * -1; crop.Width += left; left = 0; }
113  if (top < 0) { crop.Y += top * -1; crop.Height += top; top = 0; }
114  if (right < 0) { crop.Width += right; right = 0; }

118  if (left == 0 && top == 0 && right == 0 && bottom == 0) {
