206  public T get<T>(string selector, T defaultValue) where T : struct, IConvertible
207  {
208  //if (!typeof(T).IsEnum) throw new ArgumentException("T must be an enumerated type");
209 
210  string value = get(selector, null);
211  if (value == null) return defaultValue;
212  else value = value.Trim();
213  try {
214  return (T)Enum.Parse(typeof(T), value, true);
215  } catch (ArgumentException) {
216  //Build a list of valid names for the enumeration
217  string[] validNames = Enum.GetNames(typeof(T));
218  string valid = "";
219  for (int i = 0; i < validNames.Length; i++)
220  valid += (i == validNames.Length -1) ? ", and " + validNames[i] : ((i != 0) ? ", " : "") + validNames[i];
221 
222  configurationSectionIssues.AcceptIssue(new Issue("Failed to parse " + selector + ". Invalid value \"" + value + "\".", "Valid values are " + valid, IssueSeverity.ConfigurationError));
223  return defaultValue;
224  }
