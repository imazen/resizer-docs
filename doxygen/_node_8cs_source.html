209  public ICollection<Node> queryUncached(string selector) {
210  if (children == null || children.Count == 0) return null;
211 
212  selector = selector.Trim('.'); //Trim leading and trailing dots
213 
214  int nextDot = selector.IndexOf('.');
215  //Get the first item
216  string nextBit = (nextDot > -1) ? selector.Substring(0, nextDot) : selector;
217  //Get the remainder of the query
218  string remainder = (nextDot > -1) ? selector.Substring(nextDot + 1) : null;
219 
220 
221  List<Node> results = null;
222  foreach (Node n in children) {
223  if (n.Name.Equals(nextBit, StringComparison.OrdinalIgnoreCase)) {
224  //If this is the last part of the query, add results directly
225  if (string.IsNullOrEmpty(remainder)) {
226  if (results == null) results = new List<Node>();
227  results.Add(n);
228  } else {
229  //Execute subquery and add results
230  ICollection<Node> subQueryResults = n.queryUncached(remainder);
231  if (subQueryResults != null) {
232  if (results == null) results = new List<Node>();
233  results.AddRange(subQueryResults);
234  }
235  }
236  }
237  }
238  return results;
239  }
