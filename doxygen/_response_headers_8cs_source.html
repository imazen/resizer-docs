00036         public static void DefaultApplyToResponseMethod(IResponseHeaders headers, HttpContext context) {
00037             //The check against the current time is because  files served from another server may have a modified date in the future, if the clocks are not synchronized.
00038             //ASP.NET incorrectly blocks an future modified date from being sent, with an ArgumentOutOfRangeException
00039             DateTime? utc = headers.LastModified.ToUniversalTime();
00040             if (headers.LastModified == DateTime.MinValue || utc == DateTime.MinValue || utc >= DateTime.UtcNow) utc = null;
00041 
00042             //We can only update headers in Integrated mode. Otherwise we have to clear them to be able to change them.
00043             if (!HttpRuntime.UsingIntegratedPipeline) context.Response.ClearHeaders();
00044             else foreach (string key in headers.DefaultHeaders)
00045                     context.Response.Headers[key] = headers.DefaultHeaders[key];
00046 
00047             //Set the Content-Type: header
00048             if (headers.ContentType != null) context.Response.ContentType = headers.ContentType;
00049             //Sets the Expires: header
00050             if (headers.Expires != DateTime.MinValue) context.Response.Cache.SetExpires(headers.Expires);
00051             //Sets the Last-Modifed: header
00052             if (utc != null) context.Response.Cache.SetLastModified(utc.Value);
00053             //Valid until expires (I.e, ignore refresh requests)
00054             context.Response.Cache.SetValidUntilExpires(headers.ValidUntilExpires);
00055             //Omit the Vary: * 
00056             context.Response.Cache.SetOmitVaryStar(headers.SuppressVaryHeader);
00057             //Add dependencies to the server cache
00058             foreach (CacheDependency d in headers.ServerCacheDependencies)
00059                 context.Response.AddCacheDependency(d);
00060 
00061             //Set Cache-Control: header
00062             context.Response.Cache.SetCacheability(headers.CacheControl);
00063 
00064             //Set ETag in a asp.net friendly manner.
00065             if (headers.Headers["ETag"] != null) {
00066                 context.Response.Cache.SetETag(headers.Headers["ETag"]);
00067                 headers.Headers.Remove("ETag");
00068             }
00069 
00070             //Apply new headers
00071             if (HttpRuntime.UsingIntegratedPipeline) {
00072                 foreach (string key in headers.Headers)
00073                     context.Response.Headers[key] = headers.Headers[key];
00074             } else {
00075                 //Merge defaults and final headers
00076                 NameValueCollection merged = new NameValueCollection(headers.DefaultHeaders);
00077                 foreach (string key in headers.Headers) merged[key] = headers.Headers[key];
00078                 //Append each, remember ASP.NET Classic will reject those already present.
00079                 foreach (string key in merged)
00080                     context.Response.AppendHeader(key, merged[key]);
00081 
00082             }
00083 
00084         }
