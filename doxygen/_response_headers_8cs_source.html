00033         public static void DefaultApplyToResponseMethod(IResponseHeaders headers, HttpContext context) {
00034             
00035             //Apply defaults
00036             foreach (string key in headers.DefaultHeaders)
00037                 context.Response.Headers[key] = headers.DefaultHeaders[key];
00038 
00039             //Set the Content-Type: header
00040             if (headers.ContentType != null) context.Response.ContentType = headers.ContentType;
00041             //Sets the Expires: header
00042             if (headers.Expires != DateTime.MinValue) context.Response.Cache.SetExpires(headers.Expires);
00043             //Sets the Last-Modifed: header
00044             //The check against the current time is because  files served from another server may have a modified date in the future, if the clocks are not synchronized.
00045             //ASP.NET incorrectly blocks an future modified date from being sent, with an ArgumentOutOfRangeException
00046             if (headers.LastModified != DateTime.MinValue && headers.LastModified < DateTime.UtcNow) context.Response.Cache.SetLastModified(headers.LastModified);
00047 
00048             //Valid until expires (I.e, ignore refresh requests)
00049             context.Response.Cache.SetValidUntilExpires(headers.ValidUntilExpires);
00050 
00051             //Vary by the querystring
00052             //context.Response.Cache.VaryByParams["*"] = true;
00053 
00054 
00055             //Omit the Vary: * 
00056             context.Response.Cache.SetOmitVaryStar(headers.SuppressVaryHeader);
00057             //Add dependencies to the server cache
00058             foreach (CacheDependency d in headers.ServerCacheDependencies)
00059                 context.Response.AddCacheDependency(d);
00060 
00061             //Set Cache-Control: header
00062             context.Response.Cache.SetCacheability(headers.CacheControl); 
00063             
00064             //Apply new headers
00065             foreach (string key in headers.Headers)
00066                 context.Response.Headers[key] = headers.Headers[key];
00067         }
