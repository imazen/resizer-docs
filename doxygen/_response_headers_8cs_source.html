36  public static void DefaultApplyToResponseMethod(IResponseHeaders headers, HttpContext context) {
37  //The check against the current time is because files served from another server may have a modified date in the future, if the clocks are not synchronized.
38  //ASP.NET incorrectly blocks an future modified date from being sent, with an ArgumentOutOfRangeException
39  DateTime? utc = headers.LastModified.ToUniversalTime();
40  if (headers.LastModified == DateTime.MinValue || utc == DateTime.MinValue || utc >= DateTime.UtcNow) utc = null;
41 
42  //We can only update headers in Integrated mode. Otherwise we have to clear them to be able to change them.
43  if (!HttpRuntime.UsingIntegratedPipeline) context.Response.ClearHeaders();
44  else foreach (string key in headers.DefaultHeaders)
45  context.Response.Headers[key] = headers.DefaultHeaders[key];
46 
47  //Set the Content-Type: header
48  if (headers.ContentType != null) context.Response.ContentType = headers.ContentType;
49  //Sets the Expires: header
50  if (headers.Expires != DateTime.MinValue) context.Response.Cache.SetExpires(headers.Expires);
51  //Sets the Last-Modifed: header
52  if (utc != null) context.Response.Cache.SetLastModified(utc.Value);
53  //Valid until expires (I.e, ignore refresh requests)
54  context.Response.Cache.SetValidUntilExpires(headers.ValidUntilExpires);
55  //Omit the Vary: *
56  context.Response.Cache.SetOmitVaryStar(headers.SuppressVaryHeader);
57  //Add dependencies to the server cache
58  foreach (CacheDependency d in headers.ServerCacheDependencies)
59  context.Response.AddCacheDependency(d);
60 
61  //Set Cache-Control: header
62  context.Response.Cache.SetCacheability(headers.CacheControl);
63 
64  //Set ETag in a asp.net friendly manner.
65  if (headers.Headers["ETag"] != null) {
66  context.Response.Cache.SetETag(headers.Headers["ETag"]);
67  headers.Headers.Remove("ETag");
68  }
69 
70  //Apply new headers
71  if (HttpRuntime.UsingIntegratedPipeline) {
72  foreach (string key in headers.Headers)
73  context.Response.Headers[key] = headers.Headers[key];
74  } else {
75  //Merge defaults and final headers
76  NameValueCollection merged = new NameValueCollection(headers.DefaultHeaders);
77  foreach (string key in headers.Headers) merged[key] = headers.Headers[key];
78  //Append each, remember ASP.NET Classic will reject those already present.
79  foreach (string key in merged)
80  context.Response.AppendHeader(key, merged[key]);
81 
82  }
83
