167  private void DownloadFile(Dependency d, StringBuilder message) {
168  try {
169  if (File.Exists(d.DestPath)) File.Delete(d.DestPath);
170  d.Client.DownloadFile(d.Url, d.DestPath);
171  lock (message) {
172  if (d.Exists) message.AppendLine(d.Name + " reported size of " + d.ExistingLength + " instead of expected " + d.ExpectedLength + " Re-downloaded from " + d.Url);
173  else message.AppendLine(d.Name + " was missing. Downloaded from " + d.Url);
174  }
175  long downloadLength = new FileInfo(d.DestPath).Length;
176  if (downloadLength != d.ExpectedLength)
177  this.AcceptIssue(new Issues.Issue("Infinite dependency download! Expected file length " + d.ExpectedLength + " Downloaded file length " + downloadLength + ". Please notify support that the dependency manifest for " + d.RequestingAssembly + " needs to be updated."));
178  } catch (Exception we) {
179  this.AcceptIssue(new Issues.Issue("Failed to download native dependency " + d.Name + " for " + d.RequestingAssembly + " from " + d.Url, we.Message, Issues.IssueSeverity.Error));
