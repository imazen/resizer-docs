13  /// Redirects image 404 errors to a querystring-specified server-local location, while maintaining querystring values so layout isn't disrupted.
14  /// For example, missingimage.jpg?404=image.jpg&amp;width=200
15  /// with the default setting &lt;image404 baseDir="~/" /&gt; will redirect to ~/image.jpg?width=200.
16  /// You may also configure 'variables', which is the reccomended approach.
17  /// Ex. &lt;image404 propertyImageDefault="~/images/nophoto.png" /&gt; and use them like so: missingimage.jpg?404=propertImageDefault?width=200 -> ~/images/nophoto.png?width=200.

33  void Pipeline_ImageMissing(System.Web.IHttpModule sender, System.Web.HttpContext context, Configuration.IUrlEventArgs e) {
34  if (!string.IsNullOrEmpty(e.QueryString["404"])) {
35  //Resolve the path to virtual or app-relative for
36  string path = resolve404Path(e.QueryString["404"]);
37  //Resolve to virtual path
38  path = Util.PathUtils.ResolveAppRelative(path);
39 
40  //Merge commands from the 404 querystring with ones from the original image.
41  ResizeSettings imageQuery = new ResizeSettings(e.QueryString);
42  imageQuery.Normalize();
43  imageQuery.Remove("404"); //Remove the 404 ref
44 
45  ResizeSettings i404Query = new ResizeSettings(Util.PathUtils.ParseQueryString(path));
46  i404Query.Normalize();
47  //Overwrite new with old
48  foreach (string key in i404Query.Keys)
49  if (key != null) imageQuery[key] = i404Query[key];
50 
51  path = PathUtils.AddQueryString(PathUtils.RemoveQueryString(path), PathUtils.BuildQueryString(imageQuery));
52  //Redirect

57  protected string resolve404Path(string path) {
58  //1 If it starts with 'http(s)://' throw an exception.
59  if (path.StartsWith("http", StringComparison.OrdinalIgnoreCase)) throw new ImageProcessingException("Image 404 redirects must be server-local. Received " + path);
60 
61  //2 If it starts with a slash, use as-is
62  if (path.StartsWith("/", StringComparison.OrdinalIgnoreCase)) return path;
63  //3 If it starts with a tilde, use as-is.
64  if (path.StartsWith("~", StringComparison.OrdinalIgnoreCase)) return path;
65  //3 If it doesn't have a slash or a period, see if it is a attribute of <image404>.
66  if (new Regex("^[a-zA-Z][a-zA-Z0-9]*$").IsMatch(path)) {
67  string val = c.get("image404." + path,null);
68  if (val != null) return val;
69  }
70  //4 Otherwise, join with image404.basedir or the application root
71  string baseDir = c.get("image404.basedir","~/");
72  path = baseDir.TrimEnd('/') + '/' + path.TrimStart('/');
73  return path;
74  }
