00057         protected string resolve404Path(string path) {
00058             //1 If it starts with 'http(s)://' throw an exception.
00059             if (path.StartsWith("http", StringComparison.OrdinalIgnoreCase)) throw new ImageProcessingException("Image 404 redirects must be server-local. Received " + path);
00060 
00061             //2 If it starts with a slash, use as-is
00062             if (path.StartsWith("/", StringComparison.OrdinalIgnoreCase)) return path;
00063             //3 If it starts with a tilde, use as-is.
00064             if (path.StartsWith("~", StringComparison.OrdinalIgnoreCase)) return path;
00065             //3 If it doesn't have a slash or a period, see if it is a attribute of <image404>.
00066             if (new Regex("^[a-zA-Z][a-zA-Z0-9]*$").IsMatch(path)) {
00067                 string val = c.get("image404." + path,null);
00068                 if (val != null) return val;
00069             }
00070             //4 Otherwise, join with image404.basedir or the application root
00071             string baseDir = c.get("image404.basedir","~/");
00072             path = baseDir.TrimEnd('/') + '/' + path.TrimStart('/');
00073             return path;
00074         }
