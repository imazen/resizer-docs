61  /// <summary>
62  /// Not for external use. Assumes 'state' is a BatchResizeSettings instance, and stores in in a class member.
63  /// Next, it completes (or fails at) the job. Throws exceptions only if (a) state is not a BatchResizeSettings instance, or (b) an exception is thrown by and handler of the JobEvent (Failed) event.
64  /// </summary>
65  /// <param name="state"></param>
66  protected internal void Work(object state)
67  {
68  s = (BatchResizeSettings)state;
69  try
70  {
71  //Time the job.
72  jobTimer = new Stopwatch();
73  jobTimer.Start();
74 
75  //1) Eliminate duplicate target filenames, sanitize, and add appropriate extensions, then mark all items as immutable
76  s.FixDuplicateFilenames();
77  s.AppendFinalExtensions();
78  s.MarkItemsImmutable();
79 
80  //2) Build dictionary based on target filenames.
81  items = BuildDict(s.files);
82 
83  //3) Verify destination folder exists, otherwise create it. (early detection of security failures)
84  CreateDestinationFolder();
85 
86  //4) Open zip file (writes to a temporary location, then moves the file when it is done)
87  using (Ionic.Zip.ZipFile z = new Ionic.Zip.ZipFile(s.destinationFile))
88  {
89  //Compress for speed.
90  z.CompressionLevel = Ionic.Zlib.CompressionLevel.BestSpeed;
91  //Fire events instead of throwing exceptions
92  z.ZipErrorAction = ZipErrorAction.InvokeErrorEvent;
93  z.ZipError +=new EventHandler<ZipErrorEventArgs>(z_ZipError);
94  z.SaveProgress +=new EventHandler<SaveProgressEventArgs>(z_SaveProgress);
95  //5) Create a list to store item resize/compression results
96  results = new List<ItemResult>();
97 
98 
99  //6) Queue the files that will be included in the archive, specifying a callback for items
100  foreach (BatchResizeItem i in s.files){
101  try{
102  ZipEntry ze = z.AddEntry(i.TargetFilename, new WriteDelegate(WriteItemCallback));
103  //Don't try to compress files we have already resized, it won't help, just slow things down and possibly use more space.
104  if (wouldResize(i)) ze.CompressionMethod = CompressionMethod.None;
105  else
106  {
107  //Set the times
108  //ze.SetEntryTimes(System.IO.File.GetLastWriteTimeUtc(i.PhysicalPath),
109  // System.IO.File.GetLastAccessTimeUtc(i.PhysicalPath),
110  // System.IO.File.GetCreationTimeUtc(i.PhysicalPath));
111  //Nevermind, not desired behavior.
112  }
113  }catch (Exception ex){
114  ItemFailed(i,ex);
115  }
116  }
117  //7) Begin proccessing the queue.
118  //As each file is written to the archive, a callback is invoked which actually provides the resized stream.
119  z.Save();
120 
121 
122  } //8) Close zip file
123 
124  jobTimer.Stop();
125 
126  //9) Fire the event that says "We're done!"
127  JobCompleted(s);
128  }
129  catch (Exception ex)
130  {
131  //Admit we failed to whoever is listening.
132  JobFailed(s,ex);
133  }
134  }

177  public static void CopyStreamTo(System.IO.Stream src, System.IO.Stream dest)
178  {
179  int size = (src.CanSeek) ? Math.Min((int)(src.Length - src.Position), 0x2000) : 0x2000;
180  byte[] buffer = new byte[size];
181  int n;
182  do
183  {
184  n = src.Read(buffer, 0, buffer.Length);
185  dest.Write(buffer, 0, n);
186  } while (n != 0);
