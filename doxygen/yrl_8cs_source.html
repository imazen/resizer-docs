00357         /// <summary>
00358         /// Returns the type of path 
00359         /// </summary>
00360         /// <param name="path"></param>
00361         /// <returns></returns>
00362         public static PathType GetPathType(string path)
00363         {
00364             if (path == null) return PathType.Invalid;
00365             if (path.Trim(new char[] { '\\', '/', '~' }).Length == 0) return PathType.Root;
00366             int firstbackslash = path.IndexOf('\\');
00367             int firstforwardslash = path.IndexOf('/');
00368             int firsttilde = path.IndexOf('~');
00369             int firstcolon = path.IndexOf(':');
00370             int firstdoubleforwardslash = path.IndexOf("//");
00371             if (firstbackslash == 0 && path[1] == '\\')
00372             {
00373                 return PathType.UNCPath;
00374             }
00375             if (firstbackslash == 0 && firstforwardslash < 0 && firstcolon < 0)
00376             {
00377                 return PathType.BackslashPath;
00378             }
00379             if (firstcolon == 1 && firstbackslash == 2)
00380             {
00381                 return PathType.DriveletterColonBackslashPath;
00382             }
00383             if (firstcolon < 0 && firstbackslash > 0 && firstforwardslash < 0)
00384             {
00385                 return PathType.PathBackslashPath;
00386             }
00387             if (firsttilde == 0 && firstforwardslash == 1)
00388             {
00389                 return PathType.TildeForwardslashPath;
00390             }
00391             if (firstforwardslash == 0 && firsttilde != 0 && firstdoubleforwardslash < 0)
00392             {
00393                 return PathType.ForwardslashPath;
00394             }
00395             if (firstcolon > 0 && firstdoubleforwardslash > 0 && firsttilde != 0)
00396             {
00397                 return PathType.ServerForwardslashPath;
00398             }
00399             if (firsttilde != 0 && firstforwardslash < 0 && firstbackslash < 0 && firstcolon < 0)
00400             {
00401                 return PathType.File;
00402             }
00403             if (firstforwardslash > 0)
00404             {
00405                 return PathType.PathForwardslashPath;
00406             }
00407             return PathType.Invalid;
00408         }
