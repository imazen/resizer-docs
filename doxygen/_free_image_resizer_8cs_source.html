56  if (!validAlg) throw new ImageProcessingException("The specified resizing filter '" + sf + "' did not match bicubic, bilinear, box, bspline, catmullrom, or lanczos.");
57 
58  //Set copy attributes
59  s.copyAttibutes.SetWrapMode(WrapMode.TileFlipXY);

62  //The minimum dimensions of the temporary bitmap.
63  SizeF targetSize = PolygonMath.getParallelogramSize(s.layout["image"]);
64  targetSize = new SizeF((float)Math.Ceiling(targetSize.Width), (float)Math.Ceiling(targetSize.Height));
65 
66  s.ApplyCropping();
67  s.EnsurePreRenderBitmap();
68 
69  //The size of the temporary bitmap.
70  //We want it larger than the size we'll use on the final copy, so we never upscale it
71  //- but we also want it as small as possible so processing is fast.
72  SizeF tempSize = PolygonMath.ScaleOutside(targetSize, s.copyRect.Size);
73  int tempWidth = (int)Math.Ceiling(tempSize.Width);
74  int tempHeight = (int)Math.Ceiling(tempSize.Height);
