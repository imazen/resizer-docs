95  /// <summary>

102  void Pipeline_PostRewrite(IHttpModule sender, HttpContext context, Configuration.IUrlEventArgs e) {
103  string prefix = vpp.VirtualFilesystemPrefix;
104 
105  // Check if prefix is within virtual file system and if there is no querystring
106  if (e.VirtualPath.StartsWith(prefix, StringComparison.OrdinalIgnoreCase) && e.QueryString.Count == 0) {
107 
108  // Strip prefix from virtual path; keep container and blob
109  string relativeBlobURL = e.VirtualPath.Substring(vPath.Length -1).Trim('/', '\\');
110 
111  // Redirect to blob
112  context.Response.Redirect(blobStorageEndpoint + relativeBlobURL);
113  }
114  }
