00040         /// <summary>
00041         /// Returns a string for the subfolder name. The bits used are from the end of the hash - this should make
00042         /// the hashes in each directory more unique, and speed up performance (8.3 filename calculations are slow when lots of files share the same first 6 chars.
00043         /// Returns null if not configured. Rounds subfolders up to the nearest power of two.
00044         /// </summary>
00045         /// <param name="hash"></param>
00046                 /// <param name="subfolders"></param>
00047         /// <returns></returns>
00048         protected string getSubfolder(byte[] hash, int subfolders) {
00049             int bits = (int)Math.Ceiling(Math.Log(subfolders, 2)); //Log2 to find the number of bits. round up.
00050             Debug.Assert(bits > 0);
00051             Debug.Assert(bits <= hash.Length * 8);
00052 
00053             byte[] subfolder = new byte[(int)Math.Ceiling((double)bits / 8.0)]; //Round up to bytes.
00054             Array.Copy(hash, hash.Length - subfolder.Length, subfolder, 0, subfolder.Length);
00055             subfolder[0] = (byte)((int)subfolder[0] >> ((subfolder.Length * 8) - bits)); //Set extra bits to 0.
00056             return Base16Encode(subfolder);
