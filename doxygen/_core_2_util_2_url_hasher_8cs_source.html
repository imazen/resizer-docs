15  /// <summary>
16  /// Builds a key for the cached version, using the hashcode of the normalized URL.
17  /// if subfolders > 1, dirSeparator will be used to separate the subfolder and the key.
18  /// No extension is appended.
19  /// I.e, a13514\124211ab132592 or 12412ababc12141
20  /// </summary>
21  /// <param name="data"></param>
22  /// <param name="subfolders"></param>
23  /// <param name="dirSeparator"></param>
24  /// <returns></returns>
25  public string hash(string data, int subfolders, string dirSeparator) {
26 
27  SHA256 h = System.Security.Cryptography.SHA256.Create();
28  byte[] hash = h.ComputeHash(new System.Text.UTF8Encoding().GetBytes(data));
29 
30  //If configured, place files in subfolders.
31  string subfolder = "";
32  if (subfolders > 1) {
33  subfolder = getSubfolder(hash, subfolders) + dirSeparator;
34  }
35 
36  //Can't use base64 hash... filesystem has case-insensitive lookup.
37  //Would use base32, but too much code to bloat the resizer. Simple base16 encoding is fine
38  return subfolder + Base16Encode(hash);
39  }
40  /// <summary>
41  /// Returns a string for the subfolder name. The bits used are from the end of the hash - this should make
42  /// the hashes in each directory more unique, and speed up performance (8.3 filename calculations are slow when lots of files share the same first 6 chars.
43  /// Returns null if not configured. Rounds subfolders up to the nearest power of two.
44  /// </summary>
45  /// <param name="hash"></param>
46  /// <param name="subfolders"></param>
47  /// <returns></returns>
48  protected string getSubfolder(byte[] hash, int subfolders) {
49  int bits = (int)Math.Ceiling(Math.Log(subfolders, 2)); //Log2 to find the number of bits. round up.
50  Debug.Assert(bits > 0);
51  Debug.Assert(bits <= hash.Length * 8);
52 
53  byte[] subfolder = new byte[(int)Math.Ceiling((double)bits / 8.0)]; //Round up to bytes.
54  Array.Copy(hash, hash.Length - subfolder.Length, subfolder, 0, subfolder.Length);
55  subfolder[0] = (byte)((int)subfolder[0] >> ((subfolder.Length * 8) - bits)); //Set extra bits to 0.
56  return Base16Encode(subfolder);
57 
58  }

61  protected string Base16Encode(byte[] bytes) {
62  StringBuilder sb = new StringBuilder(bytes.Length * 2);
63  foreach (byte b in bytes)
64  sb.Append(b.ToString("x", NumberFormatInfo.InvariantInfo).PadLeft(2, '0'));
65  return sb.ToString();
66  }
67  }
