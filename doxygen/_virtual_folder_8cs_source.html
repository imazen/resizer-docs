132  /// <summary>
133  /// Collapses any .. segments
134  /// </summary>
135  /// <param name="path"></param>
136  /// <returns></returns>
137  protected string collapsePath(string path) {
138  string oldPath = path;
139  do {
140  oldPath = path;
141  path = collapseOneLevel(oldPath);
142  } while (oldPath != path);
143  return path;

146  protected string collapseOneLevel(string path) {
147  int up = path.Length - 1;
148  do {
149  up = path.LastIndexOf(System.IO.Path.DirectorySeparatorChar + ".." + System.IO.Path.DirectorySeparatorChar, up);
150  if (up < 0) return path;
151  int prevSlash = path.LastIndexOf(System.IO.Path.DirectorySeparatorChar, up - 1);
152  if (prevSlash < 0) return path;
153  string segment = path.Substring(prevSlash + 1, up - prevSlash - 1);
154  if (segment.Equals("..", StringComparison.OrdinalIgnoreCase)) {
155  //We can't combine \..\..\, just keep looking closer to the beginning of the string. We already adjusted 'up'
156  } else if (segment.Equals(".", StringComparison.OrdinalIgnoreCase)) {
157  return path.Substring(0, prevSlash) + path.Substring(up); //Just remove \.\ sections
158  } else {
159  return path.Substring(0, prevSlash) + path.Substring(up + 3); //If it's not \.\ or \..\, remove both it and the following \..\
160  }
161  } while (up > 0);
162  return path;
163  }
