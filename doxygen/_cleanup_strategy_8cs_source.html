56  protected void LoadProperty(NameValueCollection data, string key) {
57  string value = data[key];
58  if (string.IsNullOrEmpty(value)) return;
59 
60 
61  PropertyInfo pi = this.GetType().GetProperty(key);
62 
63  if (pi.PropertyType.Equals(typeof(int))) {
64  //It's an integer property
65  int iValue = int.MinValue;
66  if (!int.TryParse(value, NumberStyles.Integer, NumberFormatInfo.InvariantInfo, out iValue)) return; //We couldn't parse a value.
67 
68  pi.SetValue(this, iValue, null);
69  } else {
70  //It's a time span property
71  //Parse the timespan (format [ws][-]{ d | d.hh:mm[:ss[.ff]] | hh:mm[:ss[.ff]] }[ws])
72  TimeSpan tValue = TimeSpan.MinValue;
73  //Culture invariant by default
74  if (!TimeSpan.TryParse(value, out tValue)) tValue = TimeSpan.MinValue;
75 
76  //Parse it as an integer number of seconds. Seconds is the default, unlike TimeSpan.TryParse which uses days.
77  int iValue = int.MinValue;
78  if (int.TryParse(value, NumberStyles.Integer, NumberFormatInfo.InvariantInfo, out iValue)) tValue = new TimeSpan(0, 0, iValue);
79 
80  if (tValue == TimeSpan.MinValue) return; //We couldn't parse a value.
81 
82  pi.SetValue(this, tValue, null);
83  }
84  }
