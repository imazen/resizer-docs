00056         protected void LoadProperty(NameValueCollection data, string key) {
00057             string value = data[key];
00058             if (string.IsNullOrEmpty(value)) return;
00059 
00060 
00061             PropertyInfo pi = this.GetType().GetProperty(key);
00062 
00063             if (pi.PropertyType.Equals(typeof(int))) {
00064                 //It's an integer property
00065                 int iValue = int.MinValue;
00066                 if (!int.TryParse(value, NumberStyles.Integer, NumberFormatInfo.InvariantInfo, out iValue)) return; //We couldn't parse a value.
00067 
00068                 pi.SetValue(this, iValue, null);
00069             } else {
00070                 //It's a time span property
00071                 //Parse the timespan (format [ws][-]{ d | d.hh:mm[:ss[.ff]] | hh:mm[:ss[.ff]] }[ws])
00072                 TimeSpan tValue = TimeSpan.MinValue;
00073                 //Culture invariant by default
00074                 if (!TimeSpan.TryParse(value, out tValue)) tValue = TimeSpan.MinValue;
00075 
00076                 //Parse it as an integer number of seconds. Seconds is the default, unlike TimeSpan.TryParse which uses days.
00077                 int iValue = int.MinValue;
00078                 if (int.TryParse(value, NumberStyles.Integer, NumberFormatInfo.InvariantInfo, out iValue)) tValue = new TimeSpan(0, 0, iValue);
00079 
00080                 if (tValue == TimeSpan.MinValue) return; //We couldn't parse a value.
00081 
00082                 pi.SetValue(this, tValue, null);
00083             }
00084         }
