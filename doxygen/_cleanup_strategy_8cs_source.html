00034             string value = data[key];
00035             if (string.IsNullOrEmpty(value)) return;

00038             TimeSpan tValue = TimeSpan.MinValue;
00039             if (!TimeSpan.TryParse(value, out tValue)) tValue = TimeSpan.MinValue;
